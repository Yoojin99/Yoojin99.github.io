계산된 프로퍼티명을 볼 것이다.

```javascript
var className = ' Class'
var obj = {}

obj.'ab cd' = 'AB CD' //안됨. 점 찍으면 뒤에 문자열이 올 수 업다.
obj['ab cd'] //로 해야 한다. 프로퍼티를 하나씩 추가할 때 많이 한다.

var obj = {
  'A' + className : 'A급' //안됨. + 연산자 떄문에
}

obj['A'+className] //은 된다.

const obj2 = {
  ['A' + className]: 'A급' //된다.
}


```
