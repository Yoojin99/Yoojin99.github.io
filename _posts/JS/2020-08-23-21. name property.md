```javascript
function a(){}
console.log(a.name) //a

const b = function(){}
console.log(b.name) //b. 익명함수 표현식

const c = function cc(){}
console.log(c.name) //cc

const d = () => {}
console.log(d.name) //d

const e = {
  om1 : function (){},
  om2 (){},
  om3: () => {}
}

e.om1.name //om1
e.om2.name //om2
e.om3.name //om3
```
